-- =========================
--  UTENTI E GESTORI
-- =========================

CREATE TABLE gestori (
    GES_COD INT AUTO_INCREMENT PRIMARY KEY,
    GES_NAME VARCHAR(100) NOT NULL
    -- eventuali altri campi aziendali
);

CREATE TABLE utenti (
    UTE_COD INT AUTO_INCREMENT PRIMARY KEY,
    UTE_USERNAME VARCHAR(50) NOT NULL UNIQUE,
    UTE_PASSWORD VARCHAR(255) NOT NULL,
    UTE_ROLE ENUM('ADMIN','GESTORE','UTENTE') NOT NULL,
    UTE_GESTORE INT NULL, -- collegamento opzionale se l’utente appartiene a un gestore
    FOREIGN KEY (UTE_GESTORE) REFERENCES gestori(GES_COD)
);

-- =========================
--  CLIENTI E LOCATION
-- =========================

CREATE TABLE clienti (
    CLI_COD INT AUTO_INCREMENT PRIMARY KEY,
    CLI_NAME VARCHAR(100) NOT NULL,
    CLI_GESTORE INT NOT NULL,
    FOREIGN KEY (CLI_GESTORE) REFERENCES gestori(GES_COD)
);

CREATE TABLE locations (
    LOC_COD INT AUTO_INCREMENT PRIMARY KEY,
    LOC_NAME VARCHAR(100) NOT NULL,
    LOC_CLI INT NOT NULL,
    FOREIGN KEY (LOC_CLI) REFERENCES clienti(CLI_COD)
);

-- =========================
--  MODELLI, PANNELLI, ESPANSIONI
-- =========================

CREATE TABLE modelli (
    MOD_COD INT AUTO_INCREMENT PRIMARY KEY,
    MOD_NAME VARCHAR(100) NOT NULL
);

CREATE TABLE pannelli (
    PAN_COD INT AUTO_INCREMENT PRIMARY KEY,
    PAN_NAME VARCHAR(100) NOT NULL,
    PAN_MOD INT NOT NULL,
    PAN_CLI INT NOT NULL,
    PAN_LOC INT NOT NULL,
    FOREIGN KEY (PAN_MOD) REFERENCES modelli(MOD_COD),
    FOREIGN KEY (PAN_CLI) REFERENCES clienti(CLI_COD),
    FOREIGN KEY (PAN_LOC) REFERENCES locations(LOC_COD)
);

CREATE TABLE espansioni (
    EXP_COD INT AUTO_INCREMENT PRIMARY KEY,
    EXP_PAN INT NOT NULL,
    EXP_MOD INT NOT NULL,
    EXP_ADDR VARCHAR(20), -- es. indirizzo i2c
    FOREIGN KEY (EXP_PAN) REFERENCES pannelli(PAN_COD),
    FOREIGN KEY (EXP_MOD) REFERENCES modelli(MOD_COD)
);

-- =========================
--  PORTE
-- =========================

CREATE TABLE porte_modello (
    PMO_COD INT AUTO_INCREMENT PRIMARY KEY,
    PMO_MOD INT NOT NULL,
    PMO_EXP INT NULL,
    PMO_NAME VARCHAR(50) NOT NULL,
    PMO_TYPE ENUM('DIGITAL_IN','DIGITAL_OUT','ANALOG_IN','ANALOG_OUT') NOT NULL,
    PMO_RANGE_MIN INT DEFAULT 0,
    PMO_RANGE_MAX INT DEFAULT 1,
    FOREIGN KEY (PMO_MOD) REFERENCES modelli(MOD_COD),
    FOREIGN KEY (PMO_EXP) REFERENCES espansioni(EXP_COD)
);

CREATE TABLE porte_pannello (
    PPA_COD INT AUTO_INCREMENT PRIMARY KEY,
    PPA_PAN INT NOT NULL,
    PPA_PMO INT NOT NULL,
    FOREIGN KEY (PPA_PAN) REFERENCES pannelli(PAN_COD),
    FOREIGN KEY (PPA_PMO) REFERENCES porte_modello(PMO_COD)
);

-- =========================
--  GRUPPI
-- =========================

CREATE TABLE gruppi (
    GRP_COD INT AUTO_INCREMENT PRIMARY KEY,
    GRP_PAN INT NOT NULL,
    GRP_NAME VARCHAR(100) NOT NULL,
    FOREIGN KEY (GRP_PAN) REFERENCES pannelli(PAN_COD)
);

CREATE TABLE gruppi_porte (
    GPO_COD INT AUTO_INCREMENT PRIMARY KEY,
    GPO_GRP INT NOT NULL,
    GPO_PPA INT NOT NULL,
    FOREIGN KEY (GPO_GRP) REFERENCES gruppi(GRP_COD),
    FOREIGN KEY (GPO_PPA) REFERENCES porte_pannello(PPA_COD)
);

-- =========================
--  SCHEDULE
-- =========================

CREATE TABLE schedule (
    SCH_COD INT AUTO_INCREMENT PRIMARY KEY,
    SCH_PAN INT NOT NULL,
    SCH_NAME VARCHAR(100) NOT NULL,
    SCH_CRON VARCHAR(100) NOT NULL,
    FOREIGN KEY (SCH_PAN) REFERENCES pannelli(PAN_COD)
);

CREATE TABLE schedule_gruppi (
    SGR_COD INT AUTO_INCREMENT PRIMARY KEY,
    SGR_SCH INT NOT NULL,
    SGR_GRP INT NOT NULL,
    FOREIGN KEY (SGR_SCH) REFERENCES schedule(SCH_COD),
    FOREIGN KEY (SGR_GRP) REFERENCES gruppi(GRP_COD)
);

-- =========================
--  CONFIG PORTE OUTPUT
-- =========================

CREATE TABLE porte_output_config (
    POC_COD INT AUTO_INCREMENT PRIMARY KEY,
    POC_PPA INT NOT NULL,
    POC_MODE ENUM('MANUALE','AUTOMATICO') NOT NULL,
    POC_VALUE INT NULL,
    POC_GRP INT NULL,
    FOREIGN KEY (POC_PPA) REFERENCES porte_pannello(PPA_COD),
    FOREIGN KEY (POC_GRP) REFERENCES gruppi(GRP_COD)
);
